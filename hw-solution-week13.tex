                          \left[\frac{x_{(n)}+x_{(1)}}{2}-\frac{\alpha^{\frac{1}{1-n}}-1}{2}\left(x_{(n)}-x_{(1)}\right),\frac{x_{(n)}+x_{(1)}}{2}+\frac{\alpha^{\frac{1}{1-n}}-1}{2}\left(x_{(n)}-x_{(1)}\right)\right]
                      \end{equation*}
            \end{enumerate}
        \end{proof}
\end{enumerate}
